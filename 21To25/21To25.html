                                       public int[] day = new int[7];
                                       public Counter() {
                                           
                                       }
                                       
                                       public short getExercises() {
                                           return exercises;
                                       }
                                       
                                       public void increment() {
                                           exercises++;
                                       }
                                       
                                       public void increment(int total) {
                                           exercises += total;
                                       }
                                       
                                       /**
                                        * resets the count of exercises done that day
                                        */
                                       public void resetExercises() {
                                           exercises = 0;
                                       }
                                       
                                       /**
                                        * reset the week data structure
                                        */
                                       public void resetWeek() {
                                           for(int i = 0; i < day.length; i++)
                                               day[i] = 0;
                                       }
                                       
                                       /**
                                        * 0 : sunday
                                        * 1 : monday
                                        * .
                                        * .
                                        * .
                                        * .
                                        * 6 : saturday
                                        * 
                                        * @param dayOfWeek
                                        */
                                       public void setDaysTotal(int dayOfWeek) {
                                           final short current = getExercises();
                                           day[dayOfWeek] = current;
                                       }
