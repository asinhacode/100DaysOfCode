                        # translate
                        # chane any vowels to letter g
                        def translate(phrase):
                            changed = ""
                            for letter in phrase:
                                # if letter in "AEIOUaeiou":
                                if letter.lower() in "aeiou": # change it to lower case

                                    changed += "g"
                                else:
                                    changed += letter
                            return changed

                        print(translate("Anupam"))
