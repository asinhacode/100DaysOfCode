  3. Insertion Sort

    4. Wrote the pseudocode and coded in java. Just the beginning.

                        
                        package sorting;

import java.util.Arrays;

public class Insertion {

	public static void main(String[] args) {
		int[] mixed = {3,2,4,3,1,2,54,7,1,0,-4};
		
		insertion(mixed);
		
	}
	
	/**
	 * Pseudo code:
	 * i <- 1
	 * while i < length(A)
		 * 	j = i;
		 * 	while j > 0 and A[j - 1] > A[j]
		 * 		swap
		 * 		j--
	 * 		end while
	 * 	i++
	 * end while
	 * @param mixed
	 */
	public static void insertion(int[] mixed) {
		int i = 1;
		
		while(i < mixed.length) {
			int j = i;
			while (j > 0 && mixed[j - 1] > mixed[j]) {
				int temp = mixed[j - 1];
				mixed[j - 1] = mixed[j];
				mixed[j] = temp;
				j--;
			}
			i++;
		}
		
		System.out.println(Arrays.toString(mixed));
	}

}

  3. LINKED LIST 
    •                             
                                    /**
                                    * 	addFirst(data)
                                    * ----------------
                                    * 	newest = Node(data)
                                    * 	newest.next = head
                                    * 	head = newest
                                    * 	size++
                                    * 
                                    * 
                                    * addLast(data)
                                    * -----------------
                                    * 	newest = Node(data)
                                    * 	newest.next = null;
                                    * 	tail.next = newest;
                                    * 	tail = newest
                                    * 	size++
                                    * 
                                    * 
                                    * removeFirst()
                                    * ------------------
                                    * 	if head == null
                                    * 		empty list
                                    * 
                                    * 	head = head.next
                                    * 	size--;
                                    */
                                   package linkedList;
                                   
                                   public class SinglyLinkedList {
                                   
                                       // private node class
                                       private static class Node{
                                           private E element;
                                           private Node next;
                                           
                                           public Node(E e, Node n) {
                                               element = e;
                                               next = n;
                                           }
                                           
                                           public E getElement() {
                                               return element;
                                           }
                                           
                                           public Node getNext(){
                                               return next;
                                           }
                                           
                                           public void setNext(Node n) {
                                               next = n;
                                           }
                                       } // end of private class
                                       
                                       
                                       private Node head = null;
                                       private Node tail = null;
                                       
                                       private int size = 0;
                                       
                                       // constructors
                                       public SinglyLinkedList() {
                                           
                                       }
                                       
                                       public int size() {
                                           return size;
                                           
                                       }
                                       
                                       public boolean isEmpty() {
                                           return size == 0;
                                       }
                                       
                                       // accessor methods
                                       public E first() {
                                           if(isEmpty()) return null;
                                           
                                           return head.getElement();
                                       }
                                       
                                       public E last() {
                                           if(isEmpty()) return null;
                                   
                                           return tail.getElement();
                                       }
                                       
                                       // update methods
                                       public void addFirst(E e) {
                                           head = new Node<>(e, head);
                                           
                                           if(size == 0)
                                               tail = head;
                                           size++;
                                       }
                                       
                                       public void addLast(E e) {
                                           Node newest = new Node<>(e,null);
                                   
                                           if(isEmpty())
                                               head = newest;
                                           
                                           else
                                               tail.setNext(newest);
                                           tail = newest;
                                           size++;
                                       }
                                       
                                       public E removeFirst() {
                                           if(isEmpty()) // empty list
                                               return null;
                                           
                                           E answer = head.getElement();
                                           head = head.getNext(); // move head to next element
                                           size--;
                                           
                                           if(size == 0) // if there was only 1 element
                                               tail = null;
                                           return answer;
                                           
                                       }
                                       
                                       
                                   }

                            Given a string, compute recursively (no loops) the number of times lowercase "hi" appears in the string.


                            countHi("xxhixx") → 1
                            countHi("xhixhix") → 2
                            countHi("hi") → 1
                        

                                public int countHi(String str) {
  
                                    int count = 0;
                                    
                                    if(str.length() less than 2)
                                    return 0;
                                    
                                    else{
                                      if(str.substring(0,2).equals("hi"))
                                      return ++count + countHi(str.substring(1));
                                      
                                      else
                                      return countHi(str.substring(1));
                                    }
                                  }
